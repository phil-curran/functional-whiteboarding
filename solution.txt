const stringReducer = (string) => {
  // string item
  let output = "";
  let count = 0;

  // iterate through string and evaluate each character
  for (let i = 0; i < string.length; i++) {
    if (count !== 1) {
      output += count + string[i];
    } else {
      output += string[i];
    }
    count = 0;
  }
  return output;
};
